## 生成器内建消息传递，第一个 next 不需要传入任何参数

```js
function* foo(x) {
  var y = x * (yield);
  return y;
}
const it = foo(6);
it.next();
const res = it.next(7);
res.value; // 42
```

## step 辅助函数控制迭代器

```js
function step(gen) {
  const it = gen();
  var last;
  return () => {
    const res = it.next(last);
    last = res.value;
  };
}
```

## 生成器实现;something 对象叫迭代器，因为接口中有个 next 方法；iterable 支持一个函数，其名称为 Symbol.iterator，调用这个函数返回一个迭代器

```js
var something = (function () {
  var nextVal;
  return {
    [Symbol.iterator]: function () {
      return this;
    },
    next() {
      if (!nextVal) {
        nextVal = 1;
      } else {
        nextVal = nextVal * 3 + 6;
      }
      return { value: nextVal, done: false };
    },
  };
})();

for (var v of something) {
  console.log(v);
  if (v > 500) break;
}
// for ...of 每次迭代自动调用next，接收到done:true自动停止
```

## 生成器迭代器

```js
function* something() {
  var nextVal;
  while (true) {
    if (!nextVal) {
      nextVal = 1;
    } else {
      nextVal = nextVal * 3 + 6;
    }
    yield nextVal;
  }
}
// 使用
for (var v of something()) {
  //...
}
```

## 停止生成器

```js
function* something() {
  try {
    var nextVal;
    while (true) {
      if (nextVal === undefined) {
        nextVal = 1;
      } else {
        nextVal = 3 * nextVal + 6;
      }
      yield;
    }
  } finally {
    console.log("cleaning up");
  }
}
var it = something();
for (var v of it) {
  console.log(v);
  if (v > 500) console.log(it.return("hello world").value);
}

//  clean up
// hello world
```

## 生成器解决回调问题

```js
function foo(x, y, cb) {
  ajax("//x+y", cb);
}
function foo(x, y) {
  ajax("//x+y", (err, data) => {
    if (err) {
      it.throw(err);
    } else {
      it.next(data);
    }
  });
}
function* main() {
  try {
    var text = yield foo(11, 13); //只是暂时阻塞了生成器本身的代码
    console.log(text);
  } catch (e) {
    console.log(e);
  }
}
var it = main();
it.next();
```

## 生成器+Promise

```js
// main同上
var it = main();
var p = it.next().value;
p.then(
  (text) => it.next(text),
  (err) => it.throw(err)
);
```
